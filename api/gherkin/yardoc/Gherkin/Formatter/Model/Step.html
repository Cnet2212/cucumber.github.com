def line_range
  range = super
  if(rows)
    range = range.first..rows[-1].line
  elsif(doc_string)
    range = range.first..doc_string.line_range.last
  end
  range
end

def outline_args
  offset = 0
  name.scan(/<[^<]*>/).map do |val|
    offset = name.index(val, offset)
    Argument.new(offset, val)
  end
end
